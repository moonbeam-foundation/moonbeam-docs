=== "Ubuntu"

    bash
    curl -sL https://deb.nodesource.com/setup_16.x | sudo -E bash -

    sudo apt install -y nodejs
    

=== "MacOS"

    bash
    # 您可以使用 Homebrew (https://docs.brew.sh/Installation)
    brew install node

    # 或者您可以使用 nvm (https://github.com/nvm-sh/nvm)
    nvm install node
